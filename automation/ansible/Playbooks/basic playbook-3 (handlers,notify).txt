- name: Install Apache & PHP on Centos 7
  hosts: webservers

  tasks:
  - name: install apache
    yum:
      name: httpd
      state: installed

  - name: start and enable service apache
    service:
      name: httpd
      state: started
      enabled: yes

  - name: install php
    yum: name={{item}} state=installed
    with_items:
    - php
    - php-mysql
    notify: restart service httpd

  handlers:
  - name: restart service httpd
    service:
      name: httpd
      state: restarted

- name: Install MySQL on Centos 7
  hosts: dbservers

  tasks:
  - name: install mariadb
    yum: name={{item}} state=installed
    with_items:
    - mariadb-server
    - mariadb

  - name: start and enable service mariadb
    service:
      name: mariadb
      state: started
      enabled: yes

- name: Verify Packages
  hosts: all

  tasks:
  - name: ensure all packages installed
    yum:
      name: "{{packages}}"
    vars:
      packages:
      - httpd
      - mariadb-server
      - mariadb
      - php
      - php-mysql
