- name: Install Apache & PHP on Centos 7
  hosts: all
  become: yes
  ignore_errors: yes
  vars:
    var: sestatus
    titlemsg: 'hello ansible'
    bodymsg: 'hello world'

  tasks:
  - name: check sestatus
    command: "{{var}}"
    register: output

  - debug:
      msg: "{{output}}"

  - name: sleep for 30 sec
    command: sleep 30
    async: 40
    poll: 5
    when: ansible_os_family == "Redhat"
    tags:
       - apache

  - name: install apache
    yum:
      name: httpd
      state: installed
    when: ansible_os_family == "Redhat"
    tags:
       - apache

  - name: start and enable service apache
    service:
      name: httpd
      state: started
      enabled: yes
    when: ansible_os_family == "Redhat"
    tags:
       - apache

  - name: copy html template
    template:
      src: /root/ansible/index.j2
      dest: /var/www/html/index.html
    notify: restart service httpd
    when: ansible_os_family == "Redhat"
    tags:
       - apache

  - name: install php
    yum: name={{item}} state=installed
    with_items:
    - php
    - php-mysql
    notify: restart service httpd
    when: ansible_os_family == "Redhat"
    tags:
       - apache

  handlers:
  - name: restart service httpd
    service:
      name: httpd
      state: restarted
    when: ansible_os_family == "Redhat"
    tags:
       - apache
